############ Java example ############
import java.util.*;

public class FileGrouper {
    public static void main(String[] args) {
        List<String> files = Arrays.asList("doc1.txt", "image.png", "doc2.txt", "script.rb", "photo.png", "main.java");
        Map<String, List<String>> grouped = new HashMap<>();
        for (String file : files) {
            int i = file.lastIndexOf('.');
            if (i != -1) {
                String ext = file.substring(i + 1);
                grouped.putIfAbsent(ext, new ArrayList<>());
                grouped.get(ext).add(file);
            }
        }
        for (String ext : grouped.keySet()) {
            List<String> group = grouped.get(ext);
            Collections.sort(group);
            System.out.println(ext + ":");
            for (String f : group) {
                System.out.println("  " + f);
            }
        }
    }
}

############ Ruby example ############
files = ["doc1.txt", "image.png", "doc2.txt", "script.rb", "photo.png", "main.java"]
grouped = Hash.new { |h, k| h[k] = [] }
files.each do |file|
  ext = File.extname(file)[1..]
  grouped[ext] << file unless ext.nil?
end
grouped.each do |ext, group|
  puts "#{ext}:"
  group.sort.each do |f|
    puts "  #{f}"
  end
end
