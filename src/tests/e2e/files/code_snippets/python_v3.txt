from functools import reduce

# Sample data: list of people with name and age
people = [
    {"name": "Alice", "age": 30},
    {"name": "Bob", "age": 24},
    {"name": "Charlie", "age": 29},
    {"name": "Diana", "age": 35},
    {"name": "Eve", "age": 22}
]

# Filter people older than 25
older_than_25 = list(filter(lambda p: p["age"] > 25, people))

# Extract names
names = list(map(lambda p: p["name"], older_than_25))

# Calculate average age
total_age = reduce(lambda acc, p: acc + p["age"], older_than_25, 0)
average_age = total_age / len(older_than_25)

# Output
print("People older than 25:", names)
print("Average age:", average_age)
