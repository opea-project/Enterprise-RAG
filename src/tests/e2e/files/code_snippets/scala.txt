object FunctionalScalaExample extends App {
  case class Person(name: String, age: Int)

  val people = List(
    Person("Alice", 30),
    Person("Bob", 24),
    Person("Charlie", 29),
    Person("Diana", 35),
    Person("Eve", 22)
  )

  // Filter people older than 25
  val olderThan25 = people.filter(_.age > 25)

  // Extract names
  val names = olderThan25.map(_.name)

  // Calculate average age
  val totalAge = olderThan25.map(_.age).sum
  val averageAge = totalAge.toDouble / olderThan25.length

  // Output
  println(s"People older than 25: ${names.mkString(", ")}")
  println(f"Average age: $averageAge%.2f")
}
